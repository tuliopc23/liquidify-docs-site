# Progress

A progress indicator that shows completion progress of a task or loading state.

## Installation

```bash
bun add liquidify-react
```

## Import

```tsx
import { Progress } from "liquidify-react/ark-ui/progress";
```

## Basic Usage

Create a simple progress bar:

```tsx
import { Progress } from "liquidify-react/ark-ui/progress";

export function BasicExample() {
  return (
    <Progress.Root value={65} max={100}>
      <Progress.Label>Download Progress</Progress.Label>
      <Progress.ValueText />
      <Progress.Track>
        <Progress.Range />
      </Progress.Track>
    </Progress.Root>
  );
}
```

## Animated Progress

Show an animated progress indicator:

```tsx
import { useState, useEffect } from "react";
import { Progress } from "liquidify-react/ark-ui/progress";

export function AnimatedExample() {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      setProgress((prev) => {
        if (prev >= 100) {
          clearInterval(timer);
          return 100;
        }
        return prev + 1;
      });
    }, 50);

    return () => clearInterval(timer);
  }, []);

  return (
    <Progress.Root value={progress} max={100}>
      <Progress.Label>Loading...</Progress.Label>
      <Progress.ValueText />
      <Progress.Track>
        <Progress.Range />
      </Progress.Track>
    </Progress.Root>
  );
}
```

## File Upload Progress

Track file upload progress with detailed information:

```tsx
import { useState } from "react";
import { Progress } from "liquidify-react/ark-ui/progress";
import { Button } from "liquidify-react/button";

export function FileUploadExample() {
  const [progress, setProgress] = useState(0);
  const [isUploading, setIsUploading] = useState(false);

  const simulateUpload = () => {
    setIsUploading(true);
    setProgress(0);

    const timer = setInterval(() => {
      setProgress((prev) => {
        if (prev >= 100) {
          clearInterval(timer);
          setIsUploading(false);
          return 100;
        }
        return prev + Math.random() * 10;
      });
    }, 200);
  };

  return (
    <div>
      <Progress.Root value={progress} max={100}>
        <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "8px" }}>
          <Progress.Label>
            {isUploading ? "Uploading document.pdf..." : "Upload Complete"}
          </Progress.Label>
          <Progress.ValueText />
        </div>
        <Progress.Track>
          <Progress.Range />
        </Progress.Track>
      </Progress.Root>

      <div style={{ marginTop: "16px" }}>
        <Button onClick={simulateUpload} disabled={isUploading}>
          {isUploading ? "Uploading..." : "Start Upload"}
        </Button>
      </div>
    </div>
  );
}
```

## Multiple Progress Indicators

Show multiple progress bars for different tasks:

```tsx
import { Progress } from "liquidify-react/ark-ui/progress";

export function MultipleProgressExample() {
  const tasks = [
    { name: "Processing images", progress: 85, color: "#3b82f6" },
    { name: "Generating thumbnails", progress: 60, color: "#10b981" },
    { name: "Uploading to server", progress: 30, color: "#f59e0b" },
    { name: "Creating backup", progress: 0, color: "#ef4444" },
  ];

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: "16px" }}>
      {tasks.map((task, index) => (
        <Progress.Root key={index} value={task.progress} max={100}>
          <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "4px" }}>
            <Progress.Label>{task.name}</Progress.Label>
            <Progress.ValueText />
          </div>
          <Progress.Track>
            <Progress.Range style={{ backgroundColor: task.color }} />
          </Progress.Track>
        </Progress.Root>
      ))}
    </div>
  );
}
```

## Circular Progress

Create a circular progress indicator:

```tsx
import { Progress } from "liquidify-react/ark-ui/progress";

export function CircularProgressExample() {
  const progress = 75;
  const radius = 40;
  const circumference = 2 * Math.PI * radius;
  const strokeDasharray = circumference;
  const strokeDashoffset = circumference - (progress / 100) * circumference;

  return (
    <Progress.Root value={progress} max={100}>
      <div style={{ display: "flex", alignItems: "center", gap: "16px" }}>
        <div style={{ position: "relative" }}>
          <svg width="100" height="100" style={{ transform: "rotate(-90deg)" }}>
            <circle cx="50" cy="50" r={radius} stroke="#e5e7eb" strokeWidth="8" fill="none" />
            <circle
              cx="50"
              cy="50"
              r={radius}
              stroke="#3b82f6"
              strokeWidth="8"
              fill="none"
              strokeDasharray={strokeDasharray}
              strokeDashoffset={strokeDashoffset}
              strokeLinecap="round"
              style={{ transition: "stroke-dashoffset 0.3s ease" }}
            />
          </svg>
          <div
            style={{
              position: "absolute",
              top: "50%",
              left: "50%",
              transform: "translate(-50%, -50%)",
              fontWeight: "bold",
            }}
          >
            {progress}%
          </div>
        </div>

        <div>
          <Progress.Label>Task Progress</Progress.Label>
          <p style={{ margin: 0, color: "gray", fontSize: "14px" }}>Almost complete</p>
        </div>
      </div>
    </Progress.Root>
  );
}
```

## Indeterminate Progress

Show an indeterminate progress for unknown duration tasks:

```tsx
import { Progress } from "liquidify-react/ark-ui/progress";

export function IndeterminateExample() {
  return (
    <Progress.Root value={null}>
      <Progress.Label>Processing...</Progress.Label>
      <Progress.Track>
        <Progress.Range
          style={{
            animation: "indeterminate 2s infinite linear",
            transformOrigin: "left",
          }}
        />
      </Progress.Track>

      <style jsx>{`
        @keyframes indeterminate {
          0% {
            transform: translateX(-100%) scaleX(0.3);
          }
          50% {
            transform: translateX(0%) scaleX(0.8);
          }
          100% {
            transform: translateX(100%) scaleX(0.3);
          }
        }
      `}</style>
    </Progress.Root>
  );
}
```

## API Reference

### Progress.Root

The root container for the progress indicator.

| Prop          | Type                         | Default        | Description                                     |
| ------------- | ---------------------------- | -------------- | ----------------------------------------------- |
| `value`       | `number \| null`             | `0`            | Current progress value (null for indeterminate) |
| `max`         | `number`                     | `100`          | Maximum progress value                          |
| `min`         | `number`                     | `0`            | Minimum progress value                          |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Progress bar orientation                        |

### Progress.Label

The label describing what is progressing.

### Progress.ValueText

Displays the current progress value as text.

### Progress.Track

The background track of the progress bar.

### Progress.Range

The filled portion indicating current progress.

## Accessibility

The Progress component follows WAI-ARIA guidelines:

- Uses `role="progressbar"` for the progress element
- Implements `aria-valuemin`, `aria-valuemax`, and `aria-valuenow`
- Provides `aria-valuetext` for human-readable progress description
- Associates labels with `aria-labelledby`
- Supports indeterminate state for unknown progress
- Works with screen readers to announce progress changes

## Best Practices

- Always provide clear labels indicating what is progressing
- Use appropriate progress values and update them smoothly
- Consider using indeterminate progress for unknown durations
- Provide estimated time remaining when possible
- Use consistent styling across your application
- Ensure sufficient color contrast for accessibility
- Test with screen readers and keyboard navigation
- Consider animation performance on lower-end devices
