# ProgressLinear

A linear progress bar component that displays completion status with smooth animations. Perfect for file uploads, form completion, step-by-step processes, or any scenario requiring visual progress feedback in a horizontal bar format.

## Installation

```bash
npm install @liquidify/components
```

## Import

```tsx
import { ProgressLinear } from "@liquidify/components/progressLinear";
```

## Usage

### Basic Progress Linear

```tsx
import { useState } from "react";
import { ProgressLinear } from "@liquidify/components/progressLinear";

export default function BasicProgressLinear() {
  const [progress, setProgress] = useState(0);

  const incrementProgress = () => {
    setProgress(prev => Math.min(100, prev + 15));
  };

  const resetProgress = () => {
    setProgress(0);
  };

  return (
    <div style={{ padding: "2rem", maxWidth: "400px", margin: "0 auto" }}>
      <h3>Task Progress</h3>

      <ProgressLinear value={progress} label={`${progress}% Complete`} />

      <div style={{ marginTop: "2rem", display: "flex", gap: "1rem" }}>
        <button onClick={incrementProgress} disabled={progress >= 100}>
          Advance Progress
        </button>
        <button onClick={resetProgress}>
          Reset
        </button>
      </div>

      <p style={{ marginTop: "1rem", color: "#666", textAlign: "center" }}>
        Current progress: {progress}%
      </p>
    </div>
  );
}
```

### File Upload with Progress

```tsx
import { useState, useEffect } from "react";
import { ProgressLinear } from "@liquidify/components/progressLinear";

export default function FileUploadProgress() {
  const [uploadProgress, setUploadProgress] = useState(0);
  const [isUploading, setIsUploading] = useState(false);
  const [uploadedFile, setUploadedFile] = useState(null);

  const simulateUpload = (file) => {
    setIsUploading(true);
    setUploadProgress(0);
    setUploadedFile(file);

    const interval = setInterval(() => {
      setUploadProgress(prev => {
        if (prev >= 100) {
          clearInterval(interval);
          setIsUploading(false);
          return 100;
        }
        return prev + Math.random() * 12;
      });
    }, 200);
  };

  const handleFileSelect = (event) => {
    const file = event.target.files[0];
    if (file) {
      simulateUpload(file);
    }
  };

  return (
    <div style={{ padding: "2rem", maxWidth: "500px", margin: "0 auto" }}>
      <h3>File Upload</h3>

      <div style={{ marginBottom: "2rem" }}>
        <input
          type="file"
          onChange={handleFileSelect}
          disabled={isUploading}
          style={{ marginBottom: "1rem" }}
        />

        {isUploading && (
          <div>
            <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "0.5rem" }}>
              <span>Uploading {uploadedFile?.name}</span>
              <span>{Math.round(uploadProgress)}%</span>
            </div>
            <ProgressLinear value={uploadProgress} />
          </div>
        )}
      </div>

      {uploadedFile && !isUploading && uploadProgress === 100 && (
        <div style={{ padding: "1rem", backgroundColor: "#d4edda", borderRadius: "8px", textAlign: "center" }}>
          <h4 style={{ color: "#155724", margin: 0 }}>âœ… Upload Complete!</h4>
          <p style={{ color: "#155724", marginTop: "0.5rem" }}>
            {uploadedFile.name} has been successfully uploaded.
          </p>
        </div>
      )}
    </div>
  );
}
```

### Multi-Step Form Progress

```tsx
import { useState } from "react";
import { ProgressLinear } from "@liquidify/components/progressLinear";

export default function MultiStepFormProgress() {
  const [currentStep, setCurrentStep] = useState(0);
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    address: "",
    preferences: ""
  });

  const steps = [
    { title: "Personal Info", fields: ["name", "email"] },
    { title: "Address", fields: ["address"] },
    { title: "Preferences", fields: ["preferences"] }
  ];

  const totalSteps = steps.length;
  const progress = ((currentStep + 1) / totalSteps) * 100;

  const nextStep = () => {
    if (currentStep < totalSteps - 1) {
      setCurrentStep(currentStep + 1);
    }
  };

  const prevStep = () => {
    if (currentStep > 0) {
      setCurrentStep(currentStep - 1);
    }
  };

  const updateFormData = (field, value) => {
    setFormData(prev => ({ ...prev, [field]: value }));
  };

  const renderCurrentStep = () => {
    const step = steps[currentStep];

    return (
      <div style={{ marginTop: "2rem" }}>
        <h4>{step.title}</h4>

        {step.fields.map(field => (
          <div key={field} style={{ marginBottom: "1rem" }}>
            <label style={{ display: "block", marginBottom: "0.5rem", textTransform: "capitalize" }}>
              {field}
            </label>
            {field === "preferences" ? (
              <textarea
                value={formData[field]}
                onChange={(e) => updateFormData(field, e.target.value)}
                rows={4}
                style={{ width: "100%", padding: "0.5rem", border: "1px solid #ccc", borderRadius: "4px" }}
                placeholder="Tell us about your preferences..."
              />
            ) : (
              <input
                type={field === "email" ? "email" : "text"}
                value={formData[field]}
                onChange={(e) => updateFormData(field, e.target.value)}
                style={{ width: "100%", padding: "0.5rem", border: "1px solid #ccc", borderRadius: "4px" }}
                placeholder={`Enter your ${field}`}
              />
            )}
          </div>
        ))}
      </div>
    );
  };

  return (
    <div style={{ padding: "2rem", maxWidth: "600px", margin: "0 auto" }}>
      <h3>Registration Form</h3>

      <div style={{ marginBottom: "2rem" }}>
        <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "0.5rem" }}>
          <span>Step {currentStep + 1} of {totalSteps}</span>
          <span>{Math.round(progress)}% Complete</span>
        </div>
        <ProgressLinear value={progress} />
      </div>

      <div style={{ marginBottom: "2rem" }}>
        <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "1rem" }}>
          {steps.map((step, index) => (
            <div
              key={index}
              style={{
                flex: 1,
                textAlign: "center",
                padding: "0.5rem",
                backgroundColor: index <= currentStep ? "#007bff" : "#e9ecef",
                color: index <= currentStep ? "white" : "#666",
                borderRadius: index === 0 ? "4px 0 0 4px" : index === steps.length - 1 ? "0 4px 4px 0" : "0"
              }}
            >
              {step.title}
            </div>
          ))}
        </div>
      </div>

      {renderCurrentStep()}

      <div style={{ marginTop: "2rem", display: "flex", justifyContent: "space-between" }}>
        <button
          onClick={prevStep}
          disabled={currentStep === 0}
          style={{
            padding: "0.5rem 1rem",
            backgroundColor: currentStep === 0 ? "#6c757d" : "#6c757d",
            color: "white",
            border: "none",
            borderRadius: "4px",
            cursor: currentStep === 0 ? "not-allowed" : "pointer"
          }}
        >
          Previous
        </button>

        <button
          onClick={nextStep}
          disabled={currentStep === totalSteps - 1}
          style={{
            padding: "0.5rem 1rem",
            backgroundColor: currentStep === totalSteps - 1 ? "#28a745" : "#007bff",
            color: "white",
            border: "none",
            borderRadius: "4px",
            cursor: currentStep === totalSteps - 1 ? "pointer" : "pointer"
          }}
        >
          {currentStep === totalSteps - 1 ? "Submit" : "Next"}
        </button>
      </div>
    </div>
  );
}
```

### Loading States with Multiple Progress Bars

```tsx
import { useState, useEffect } from "react";
import { ProgressLinear } from "@liquidify/components/progressLinear";

export default function LoadingStates() {
  const [loadingStates, setLoadingStates] = useState({
    initializing: 0,
    connecting: 0,
    authenticating: 0,
    loadingData: 0
  });

  const [currentPhase, setCurrentPhase] = useState("initializing");

  useEffect(() => {
    const phases = ["initializing", "connecting", "authenticating", "loadingData"];

    phases.forEach((phase, index) => {
      setTimeout(() => {
        setCurrentPhase(phase);
        const interval = setInterval(() => {
          setLoadingStates(prev => {
            const newValue = prev[phase] + Math.random() * 15;
            if (newValue >= 100) {
              clearInterval(interval);
              return { ...prev, [phase]: 100 };
            }
            return { ...prev, [phase]: newValue };
          });
        }, 200);
      }, index * 1500);
    });
  }, []);

  const phaseLabels = {
    initializing: "Initializing application",
    connecting: "Connecting to server",
    authenticating: "Authenticating user",
    loadingData: "Loading user data"
  };

  const allComplete = Object.values(loadingStates).every(value => value === 100);

  return (
    <div style={{ padding: "2rem", maxWidth: "500px", margin: "0 auto" }}>
      <h3>Application Startup</h3>

      <div style={{ display: "flex", flexDirection: "column", gap: "1.5rem" }}>
        {Object.entries(loadingStates).map(([key, value]) => (
          <div key={key}>
            <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "0.5rem" }}>
              <span style={{ fontWeight: currentPhase === key ? "bold" : "normal" }}>
                {phaseLabels[key]}
              </span>
              <span>{Math.round(value)}%</span>
            </div>
            <ProgressLinear
              value={value}
              size={currentPhase === key ? "lg" : "md"}
            />
          </div>
        ))}
      </div>

      {allComplete && (
        <div style={{ marginTop: "2rem", padding: "1rem", backgroundColor: "#d4edda", borderRadius: "8px", textAlign: "center" }}>
          <h4 style={{ color: "#155724", margin: 0 }}>ðŸŽ‰ Application Ready!</h4>
          <p style={{ color: "#155724", marginTop: "0.5rem" }}>
            All systems have been initialized successfully.
          </p>
        </div>
      )}
    </div>
  );
}
```

## API Reference

### ProgressLinear

The linear progress bar component.

#### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `number` | `0` | The current progress value |
| `max` | `number` | `100` | The maximum progress value |
| `size` | `"xs" \| "sm" \| "md" \| "lg"` | `"md"` | The size/height of the progress bar |
| `label` | `string` | - | The label text to display |
| `children` | `ReactNode` | - | Custom content to display instead of label |
| All standard div props | - | - | Standard HTML div attributes |

## Accessibility

The ProgressLinear component includes comprehensive accessibility features:

- **ARIA Roles**: Implements proper progressbar role
- **ARIA Labels**: Automatic labeling for screen readers
- **Screen Reader Support**: Announces progress values and status
- **Keyboard Navigation**: Full keyboard support
- **High Contrast**: Compatible with high contrast themes

## Best Practices

1. **Clear Context**: Always provide context about what's being tracked
2. **Appropriate Size**: Choose size based on available space and importance
3. **Smooth Updates**: Update progress values smoothly to avoid jarring changes
4. **Completion Feedback**: Provide clear feedback when progress reaches 100%
5. **Loading States**: Use indeterminate progress for unknown durations
6. **Performance**: Avoid updating progress too frequently to prevent performance issues
7. **User Feedback**: Combine with text labels for better accessibility
8. **Consistent Styling**: Maintain consistent progress bar styling across your app
