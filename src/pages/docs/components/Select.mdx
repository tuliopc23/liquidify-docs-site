# Select

A dropdown select component that allows users to choose from a list of options.

## Installation

```bash
bun add liquidify-react
```

## Import

```tsx
import { Select } from "liquidify-react/ark-ui/select";
```

## Basic Usage

Create a simple select dropdown:

<SelectDemo client:only="react" />

```tsx
import { Select } from "liquidify-react/ark-ui/select";

export function BasicExample() {
  const options = [
    { label: "Apple", value: "apple" },
    { label: "Banana", value: "banana" },
    { label: "Orange", value: "orange" },
    { label: "Grape", value: "grape" },
  ];

  return (
    <Select.Root items={options}>
      <Select.Label>Choose a fruit</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Select an option" />
          <Select.Indicator>▼</Select.Indicator>
        </Select.Trigger>
      </Select.Control>
      <Select.Positioner>
        <Select.Content>
          {options.map((option) => (
            <Select.Item key={option.value} item={option}>
              <Select.ItemText>{option.label}</Select.ItemText>
              <Select.ItemIndicator>✓</Select.ItemIndicator>
            </Select.Item>
          ))}
        </Select.Content>
      </Select.Positioner>
    </Select.Root>
  );
}
```

## Controlled Select

Control the selected value with state:

```tsx
import { useState } from "react";
import { Select } from "liquidify-react/ark-ui/select";

export function ControlledExample() {
  const [value, setValue] = useState([]);

  const countries = [
    { label: "United States", value: "us" },
    { label: "Canada", value: "ca" },
    { label: "United Kingdom", value: "uk" },
    { label: "Germany", value: "de" },
    { label: "France", value: "fr" },
  ];

  return (
    <Select.Root
      items={countries}
      value={value}
      onValueChange={(details) => setValue(details.value)}
    >
      <Select.Label>Select your country</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Choose a country" />
          <Select.Indicator>▼</Select.Indicator>
        </Select.Trigger>
      </Select.Control>
      <Select.Positioner>
        <Select.Content>
          {countries.map((country) => (
            <Select.Item key={country.value} item={country}>
              <Select.ItemText>{country.label}</Select.ItemText>
              <Select.ItemIndicator>✓</Select.ItemIndicator>
            </Select.Item>
          ))}
        </Select.Content>
      </Select.Positioner>
      <p>Selected: {value.join(", ")}</p>
    </Select.Root>
  );
}
```

## Multi-Select

Allow multiple selections:

```tsx
import { useState } from "react";
import { Select } from "liquidify-react/ark-ui/select";

export function MultiSelectExample() {
  const [values, setValues] = useState([]);

  const skills = [
    { label: "JavaScript", value: "javascript" },
    { label: "TypeScript", value: "typescript" },
    { label: "React", value: "react" },
    { label: "Vue", value: "vue" },
    { label: "Angular", value: "angular" },
    { label: "Node.js", value: "nodejs" },
  ];

  return (
    <Select.Root
      items={skills}
      value={values}
      onValueChange={(details) => setValues(details.value)}
      multiple
    >
      <Select.Label>Select your skills</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Choose skills" />
          <Select.Indicator>▼</Select.Indicator>
        </Select.Trigger>
      </Select.Control>
      <Select.Positioner>
        <Select.Content>
          {skills.map((skill) => (
            <Select.Item key={skill.value} item={skill}>
              <Select.ItemText>{skill.label}</Select.ItemText>
              <Select.ItemIndicator>✓</Select.ItemIndicator>
            </Select.Item>
          ))}
        </Select.Content>
      </Select.Positioner>
      {values.length > 0 && <p>Selected skills: {values.join(", ")}</p>}
    </Select.Root>
  );
}
```

## Grouped Options

Organize options into groups:

```tsx
import { Select } from "liquidify-react/ark-ui/select";

export function GroupedExample() {
  const options = [
    { label: "Fruits", value: "fruits", disabled: true },
    { label: "Apple", value: "apple" },
    { label: "Banana", value: "banana" },
    { label: "Orange", value: "orange" },
    { label: "Vegetables", value: "vegetables", disabled: true },
    { label: "Carrot", value: "carrot" },
    { label: "Broccoli", value: "broccoli" },
    { label: "Spinach", value: "spinach" },
  ];

  return (
    <Select.Root items={options}>
      <Select.Label>Food Categories</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Select food" />
          <Select.Indicator>▼</Select.Indicator>
        </Select.Trigger>
      </Select.Control>
      <Select.Positioner>
        <Select.Content>
          {options.map((option) => (
            <Select.Item
              key={option.value}
              item={option}
              style={{
                fontWeight: option.disabled ? "bold" : "normal",
                opacity: option.disabled ? 0.7 : 1,
                paddingLeft: option.disabled ? "8px" : "24px",
              }}
            >
              <Select.ItemText>{option.label}</Select.ItemText>
              {!option.disabled && <Select.ItemIndicator>✓</Select.ItemIndicator>}
            </Select.Item>
          ))}
        </Select.Content>
      </Select.Positioner>
    </Select.Root>
  );
}
```

## With Search/Filter

Add search functionality to filter options:

```tsx
import { useState } from "react";
import { Select } from "liquidify-react/ark-ui/select";

export function SearchableExample() {
  const [searchTerm, setSearchTerm] = useState("");

  const allOptions = [
    { label: "Afghanistan", value: "af" },
    { label: "Albania", value: "al" },
    { label: "Algeria", value: "dz" },
    { label: "Argentina", value: "ar" },
    { label: "Australia", value: "au" },
    { label: "Austria", value: "at" },
    { label: "Belgium", value: "be" },
    { label: "Brazil", value: "br" },
    { label: "Canada", value: "ca" },
    { label: "China", value: "cn" },
  ];

  const filteredOptions = allOptions.filter((option) =>
    option.label.toLowerCase().includes(searchTerm.toLowerCase()),
  );

  return (
    <Select.Root items={filteredOptions}>
      <Select.Label>Select Country</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Search and select country" />
          <Select.Indicator>▼</Select.Indicator>
        </Select.Trigger>
      </Select.Control>
      <Select.Positioner>
        <Select.Content>
          <div style={{ padding: "8px" }}>
            <input
              type="text"
              placeholder="Search countries..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              style={{
                width: "100%",
                padding: "8px",
                border: "1px solid #e5e7eb",
                borderRadius: "4px",
              }}
            />
          </div>
          {filteredOptions.length === 0 ? (
            <div style={{ padding: "8px", color: "#6b7280" }}>No countries found</div>
          ) : (
            filteredOptions.map((option) => (
              <Select.Item key={option.value} item={option}>
                <Select.ItemText>{option.label}</Select.ItemText>
                <Select.ItemIndicator>✓</Select.ItemIndicator>
              </Select.Item>
            ))
          )}
        </Select.Content>
      </Select.Positioner>
    </Select.Root>
  );
}
```

## API Reference

### Select.Root

The root container for the select component.

| Prop            | Type                                    | Default | Description                           |
| --------------- | --------------------------------------- | ------- | ------------------------------------- |
| `items`         | `Array<{label: string, value: string}>` | `[]`    | Array of selectable options           |
| `value`         | `string[]`                              | `[]`    | Controlled selected values            |
| `defaultValue`  | `string[]`                              | `[]`    | Default selected values               |
| `onValueChange` | `(details) => void`                     | -       | Callback when selection changes       |
| `multiple`      | `boolean`                               | `false` | Whether multiple selection is allowed |
| `disabled`      | `boolean`                               | `false` | Whether the select is disabled        |
| `invalid`       | `boolean`                               | `false` | Whether the select is invalid         |

### Select.Label

The label for the select component.

### Select.Control

Container for the trigger and other controls.

### Select.Trigger

The clickable element that opens the dropdown.

### Select.ValueText

Displays the selected value(s) or placeholder.

| Prop          | Type     | Default | Description                           |
| ------------- | -------- | ------- | ------------------------------------- |
| `placeholder` | `string` | -       | Text shown when no option is selected |

### Select.Indicator

Visual indicator (usually an arrow) for the dropdown state.

### Select.Positioner

Controls positioning of the dropdown content.

### Select.Content

Container for the selectable options.

### Select.Item

Individual selectable option.

| Prop   | Type                             | Default | Description            |
| ------ | -------------------------------- | ------- | ---------------------- |
| `item` | `{label: string, value: string}` | -       | Option data (required) |

### Select.ItemText

Text content of the option.

### Select.ItemIndicator

Visual indicator showing selected state.

## Accessibility

The Select component follows WAI-ARIA guidelines:

- Uses `role="combobox"` for the trigger
- Implements `role="listbox"` and `role="option"` for dropdown
- Supports keyboard navigation with arrow keys and Enter
- Provides proper focus management
- Associates labels with `aria-labelledby`
- Implements `aria-expanded` for dropdown state
- Supports type-ahead functionality

## Best Practices

- Provide clear, descriptive labels for each option
- Use appropriate placeholder text to guide users
- Consider adding search functionality for long lists
- Group related options when appropriate
- Ensure sufficient contrast for selected states
- Test keyboard navigation and screen reader compatibility
- Use consistent styling across your application
- Consider mobile responsiveness for touch interactions
