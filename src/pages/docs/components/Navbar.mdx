# Navbar

A responsive navigation bar component that provides site navigation with automatic scroll-based styling changes. Perfect for website headers, application navigation, or any scenario requiring prominent navigation with visual feedback based on scroll position.

## Installation

```bash
npm install @liquidify/components
```

## Import

```tsx
import { Navbar } from "@liquidify/components/navbar";
```

## Usage

### Basic Navbar

```tsx
import { Navbar } from "@liquidify/components/navbar";

export default function BasicNavbar() {
  return (
    <Navbar />
  );
}
```

                  notifications.map(notification => (
                    <div
                      key={notification.id}
                      className={css({
                        px: "16px",
                        py: "12px",
                        borderBottom: "1px solid #e9ecef",
                        bg: notification.read ? "white" : "#f8f9fa",
                        cursor: "pointer",
                        _hover: { bg: "#e9ecef" },
                      })}
                      onClick={() => markAsRead(notification.id)}
                    >
                      <div className={css({
                        fontSize: "14px",
                        color: notification.read ? "#666" : "#333",
                      })}>
                        {notification.message}
                      </div>
                      <div className={css({
                        fontSize: "12px",
                        color: "#999",
                        mt: "4px",
                      })}>
                        2 hours ago
                      </div>
                    </div>
                  ))
                )}
              </div>
            )}
          </div>

          <button
            className={css({
              px: "16px",
              py: "8px",
              bg: "token(colors.primary)",
              color: "white",
              border: "none",
              borderRadius: "6px",
              cursor: "pointer",
              _hover: { bg: "token(colors.primary.dark)" },
            })}
          >
            Upgrade
          </button>
        </div>
      </div>
    </nav>
  );
}
