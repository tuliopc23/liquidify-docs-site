# Editable

Inline text editing with smooth transitions between read and edit modes. Perfect for titles, labels, and content that needs quick editing.

## Import

```tsx
import { Editable } from "liquidify-react/ark-ui/editable";
```

## Basic Usage

```tsx
export default function Example() {
  return (
    <Editable.Root placeholder="Enter text..." defaultValue="Click to edit">
      <Editable.Area>
        <Editable.Input />
        <Editable.Preview />
      </Editable.Area>

      <Editable.Control>
        <Editable.EditTrigger asChild>
          <button>
            <EditIcon />
          </button>
        </Editable.EditTrigger>
        <Editable.SubmitTrigger asChild>
          <button>
            <CheckIcon />
          </button>
        </Editable.SubmitTrigger>
        <Editable.CancelTrigger asChild>
          <button>
            <XIcon />
          </button>
        </Editable.CancelTrigger>
      </Editable.Control>
    </Editable.Root>
  );
}
```

## Auto-resize Textarea

```tsx
export default function TextareaExample() {
  return (
    <Editable.Root
      placeholder="Enter description..."
      defaultValue="This is a multi-line description that can be edited inline."
    >
      <Editable.Area>
        <Editable.Input asChild>
          <textarea style={{ resize: "none", minHeight: "60px" }} />
        </Editable.Input>
        <Editable.Preview />
      </Editable.Area>

      <Editable.Control>
        <Editable.EditTrigger asChild>
          <Button size="sm" variant="plain" tone="neutral">
            <EditIcon />
            Edit
          </Button>
        </Editable.EditTrigger>
        <Editable.SubmitTrigger asChild>
          <Button size="sm">Save</Button>
        </Editable.SubmitTrigger>
        <Editable.CancelTrigger asChild>
          <Button size="sm" variant="tinted" tone="neutral">
            Cancel
          </Button>
        </Editable.CancelTrigger>
      </Editable.Control>
    </Editable.Root>
  );
}
```

## Controlled State

```tsx
import { useState } from "react";

export default function ControlledExample() {
  const [value, setValue] = useState("Controlled editable text");
  const [editing, setEditing] = useState(false);

  return (
    <div>
      <p>Current value: {value}</p>
      <p>Editing: {editing ? "Yes" : "No"}</p>

      <Editable.Root
        value={value}
        onValueChange={setValue}
        edit={editing}
        onEditChange={setEditing}
      >
        <Editable.Area>
          <Editable.Input />
          <Editable.Preview />
        </Editable.Area>

        <Editable.Control>
          <Editable.EditTrigger asChild>
            <Button variant="plain" tone="neutral">Edit</Button>
          </Editable.EditTrigger>
          <Editable.SubmitTrigger asChild>
            <Button>Save</Button>
          </Editable.SubmitTrigger>
          <Editable.CancelTrigger asChild>
            <Button variant="tinted" tone="neutral">Cancel</Button>
          </Editable.CancelTrigger>
        </Editable.Control>
      </Editable.Root>
    </div>
  );
}
```

## Validation

```tsx
import { useState } from "react";

export default function ValidationExample() {
  const [value, setValue] = useState("Valid text");
  const [error, setError] = useState("");

  const handleSubmit = (details) => {
    if (details.value.length < 3) {
      setError("Text must be at least 3 characters");
      return;
    }
    setError("");
    setValue(details.value);
  };

  return (
    <Editable.Root value={value} onValueCommit={handleSubmit}>
      <Editable.Area>
        <Editable.Input
          style={{
            borderColor: error ? "red" : undefined,
          }}
        />
        <Editable.Preview />
      </Editable.Area>

      {error && (
        <div style={{ color: "red", fontSize: "0.875rem", marginTop: "0.25rem" }}>{error}</div>
      )}

      <Editable.Control>
        <Editable.EditTrigger asChild>
          <Button variant="plain" tone="neutral">Edit</Button>
        </Editable.EditTrigger>
        <Editable.SubmitTrigger asChild>
          <Button>Save</Button>
        </Editable.SubmitTrigger>
        <Editable.CancelTrigger asChild>
          <Button variant="tinted" tone="neutral">Cancel</Button>
        </Editable.CancelTrigger>
      </Editable.Control>
    </Editable.Root>
  );
}
```

## Double-click to Edit

```tsx
export default function DoubleClickExample() {

        <Editable.Preview />
      </Editable.Area>
    </Editable.Root>
  );
}
```

## API Reference

### Root Props

- `value`: `string` - Controlled value
- `defaultValue`: `string` - Default value (uncontrolled)
- `placeholder`: `string` - Placeholder text
- `edit`: `boolean` - Controlled edit state
- `activationMode`: `"focus" | "dblclick" | "none"` - How to enter edit mode
- `submitMode`: `"enter" | "blur" | "none"` - How to submit changes
- `onValueChange`: `(value: string) => void` - Value change handler
- `onValueCommit`: `(details: {value: string}) => void` - Commit handler

### Accessibility

- Full keyboard navigation
- Screen reader announcements for mode changes
- Proper ARIA labels and roles
- Focus management between modes
