# DatePicker

Calendar-based date selection with keyboard navigation and localization support. Perfect for forms, scheduling, and date range selection.

## Import

```tsx
import { DatePicker } from "liquidify-react/date-picker"
```

## Basic Usage

```tsx
export default function Example() {
	return (
		<DatePicker.Root>
			<DatePicker.Label>Select Date</DatePicker.Label>
			<DatePicker.Control>
				<DatePicker.Input />
				<DatePicker.Trigger>
					<CalendarIcon />
				</DatePicker.Trigger>
			</DatePicker.Control>
			
			<DatePicker.Positioner>
				<DatePicker.Content>
					<DatePicker.View view="day">
						<DatePicker.Context>
							{(datePicker) => (
								<>
									<DatePicker.ViewControl>
										<DatePicker.PrevTrigger>
											<ChevronLeftIcon />
										</DatePicker.PrevTrigger>
										<DatePicker.ViewTrigger>
											<DatePicker.RangeText />
										</DatePicker.ViewTrigger>
										<DatePicker.NextTrigger>
											<ChevronRightIcon />
										</DatePicker.NextTrigger>
									</DatePicker.ViewControl>
									
									<DatePicker.Table>
										<DatePicker.TableHead>
											<DatePicker.TableRow>
												{datePicker.weekDays.map((weekDay, id) => (
													<DatePicker.TableHeader key={id}>
														{weekDay.narrow}
													</DatePicker.TableHeader>
												))}
											</DatePicker.TableRow>
										</DatePicker.TableHead>
										<DatePicker.TableBody>
											{datePicker.weeks.map((week, id) => (
												<DatePicker.TableRow key={id}>
													{week.map((day, id) => (
														<DatePicker.TableCell key={id} value={day}>
															<DatePicker.TableCellTrigger>
																{day.day}
															</DatePicker.TableCellTrigger>
														</DatePicker.TableCell>
													))}
												</DatePicker.TableRow>
											))}
										</DatePicker.TableBody>
									</DatePicker.Table>
								</>
							)}
						</DatePicker.Context>
					</DatePicker.View>
				</DatePicker.Content>
			</DatePicker.Positioner>
		</DatePicker.Root>
	)
}
```

## Controlled State

```tsx
import { useState } from "react"

export default function ControlledExample() {
	const [value, setValue] = useState([])
	
	return (
		<div>
			<p>Selected: {value.length > 0 ? value[0]?.toString() : "None"}</p>
			<DatePicker.Root value={value} onValueChange={setValue}>
				<DatePicker.Control>
					<DatePicker.Input />
					<DatePicker.Trigger>
						<CalendarIcon />
					</DatePicker.Trigger>
				</DatePicker.Control>
				
				<DatePicker.Positioner>
					<DatePicker.Content>
						{/* Calendar implementation */}
					</DatePicker.Content>
				</DatePicker.Positioner>
			</DatePicker.Root>
		</div>
	)
}
```

## With Constraints

```tsx
export default function ConstraintsExample() {
	const today = new Date()
	const minDate = new Date(today.getFullYear(), today.getMonth(), today.getDate())
	const maxDate = new Date(today.getFullYear() + 1, today.getMonth(), today.getDate())
	
	return (
		<DatePicker.Root 
			min={minDate}
			max={maxDate}
		>
			<DatePicker.Label>Future Dates Only</DatePicker.Label>
			<DatePicker.Control>
				<DatePicker.Input />
				<DatePicker.Trigger>
					<CalendarIcon />
				</DatePicker.Trigger>
			</DatePicker.Control>
			
			<DatePicker.Positioner>
				<DatePicker.Content>
					{/* Calendar implementation with constraints */}
				</DatePicker.Content>
			</DatePicker.Positioner>
		</DatePicker.Root>
	)
}
```

## Custom Format

```tsx
export default function FormatExample() {
	return (
		<DatePicker.Root 
			format={(date) => {
				return new Intl.DateTimeFormat("en-US", {
					weekday: "long",
					year: "numeric", 
					month: "long",
					day: "numeric"
				}).format(date)
			}}
		>
			<DatePicker.Control>
				<DatePicker.Input placeholder="Select a date..." />
				<DatePicker.Trigger>
					<CalendarIcon />
				</DatePicker.Trigger>
			</DatePicker.Control>
			
			<DatePicker.Positioner>
				<DatePicker.Content>
					{/* Calendar implementation */}
				</DatePicker.Content>
			</DatePicker.Positioner>
		</DatePicker.Root>
	)
}
```

## API Reference

### Root Props
- `value`: `Date[]` - Selected date(s)
- `defaultValue`: `Date[]` - Default selection
- `selectionMode`: `"single" | "multiple" | "range"` - Selection behavior
- `min`: `Date` - Minimum selectable date
- `max`: `Date` - Maximum selectable date
- `locale`: `string` - Locale for formatting
- `onValueChange`: `(value: Date[]) => void` - Selection handler

### Accessibility
- Full keyboard navigation with Arrow keys
- Screen reader announcements for dates
- ARIA calendar role and properties
- Focus management within calendar grid
