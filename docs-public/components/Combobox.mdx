# Combobox

Searchable dropdown with autocomplete functionality. Combines text input with a dropdown list for efficient selection from large datasets.

## Import

```tsx
import { Combobox } from "liquidify-react/combobox"
```

## Basic Usage

```tsx
export default function Example() {
	const items = [
		{ label: "React", value: "react" },
		{ label: "Vue", value: "vue" },
		{ label: "Angular", value: "angular" },
		{ label: "Svelte", value: "svelte" }
	]
	
	return (
		<Combobox.Root items={items}>
			<Combobox.Label>Choose Framework</Combobox.Label>
			<Combobox.Control>
				<Combobox.Input placeholder="Search frameworks..." />
				<Combobox.Trigger>
					<ChevronDownIcon />
				</Combobox.Trigger>
			</Combobox.Control>
			
			<Combobox.Positioner>
				<Combobox.Content>
					{items.map(item => (
						<Combobox.Item key={item.value} item={item}>
							<Combobox.ItemText>{item.label}</Combobox.ItemText>
							<Combobox.ItemIndicator>
								<CheckIcon />
							</Combobox.ItemIndicator>
						</Combobox.Item>
					))}
				</Combobox.Content>
			</Combobox.Positioner>
		</Combobox.Root>
	)
}
```

## With Filtering

```tsx
import { useState, useMemo } from "react"

export default function FilteringExample() {
	const allItems = [
		{ label: "Apple", value: "apple" },
		{ label: "Banana", value: "banana" },
		{ label: "Cherry", value: "cherry" },
		{ label: "Date", value: "date" },
		{ label: "Elderberry", value: "elderberry" }
	]
	
	const [inputValue, setInputValue] = useState("")
	
	const filteredItems = useMemo(() => {
		return allItems.filter(item =>
			item.label.toLowerCase().includes(inputValue.toLowerCase())
		)
	}, [inputValue])
	
	return (
		<Combobox.Root 
			items={filteredItems}
			inputValue={inputValue}
			onInputValueChange={setInputValue}
		>
			<Combobox.Label>Select Fruit</Combobox.Label>
			<Combobox.Control>
				<Combobox.Input placeholder="Type to search..." />
				<Combobox.Trigger>
					<SearchIcon />
				</Combobox.Trigger>
			</Combobox.Control>
			
			<Combobox.Positioner>
				<Combobox.Content>
					{filteredItems.length === 0 ? (
						<div style={{ padding: "1rem", textAlign: "center", color: "#666" }}>
							No fruits found
						</div>
					) : (
						filteredItems.map(item => (
							<Combobox.Item key={item.value} item={item}>
								<Combobox.ItemText>{item.label}</Combobox.ItemText>
							</Combobox.Item>
						))
					)}
				</Combobox.Content>
			</Combobox.Positioner>
		</Combobox.Root>
	)
}
```

## Controlled State

```tsx
import { useState } from "react"

export default function ControlledExample() {
	const items = [
		{ label: "Small", value: "sm" },
		{ label: "Medium", value: "md" },
		{ label: "Large", value: "lg" }
	]
	
	const [value, setValue] = useState("")
	
	return (
		<div>
			<p>Selected: {value || "None"}</p>
			<Combobox.Root 
				items={items}
				value={value}
				onValueChange={setValue}
			>
				<Combobox.Control>
					<Combobox.Input placeholder="Choose size..." />
					<Combobox.Trigger>
						<ChevronDownIcon />
					</Combobox.Trigger>
				</Combobox.Control>
				
				<Combobox.Positioner>
					<Combobox.Content>
						{items.map(item => (
							<Combobox.Item key={item.value} item={item}>
								<Combobox.ItemText>{item.label}</Combobox.ItemText>
							</Combobox.Item>
						))}
					</Combobox.Content>
				</Combobox.Positioner>
			</Combobox.Root>
		</div>
	)
}
```

## API Reference

### Root Props
- `items`: `Array<{label: string, value: string}>` - Available options
- `value`: `string | string[]` - Selected value(s)
- `defaultValue`: `string | string[]` - Default selection
- `multiple`: `boolean` - Allow multiple selection
- `inputValue`: `string` - Current input text
- `onValueChange`: `(value: string | string[]) => void` - Selection handler
- `onInputValueChange`: `(value: string) => void` - Input change handler

### Accessibility
- Full keyboard navigation with Arrow keys
- Screen reader announcements
- ARIA combobox role and states
- Focus management between input and options
