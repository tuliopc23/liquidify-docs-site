# ColorPicker

Interactive color selection with multiple input methods. Supports hex, RGB, HSL formats with visual swatches and precise value controls.

## Import

```tsx
import { ColorPicker } from "liquidify-react/color-picker"
```

## Basic Usage

```tsx
export default function Example() {
	return (
		<ColorPicker.Root defaultValue="#3b82f6">
			<ColorPicker.Label>Choose Color</ColorPicker.Label>
			<ColorPicker.Control>
				<ColorPicker.ChannelInput channel="hex" />
				<ColorPicker.Trigger>
					<ColorPicker.Swatch />
				</ColorPicker.Trigger>
			</ColorPicker.Control>
			
			<ColorPicker.Positioner>
				<ColorPicker.Content>
					<ColorPicker.Area>
						<ColorPicker.AreaBackground />
						<ColorPicker.AreaThumb />
					</ColorPicker.Area>
					
					<ColorPicker.ChannelSlider channel="hue">
						<ColorPicker.ChannelSliderTrack />
						<ColorPicker.ChannelSliderThumb />
					</ColorPicker.ChannelSlider>
					
					<ColorPicker.ChannelSlider channel="alpha">
						<ColorPicker.ChannelSliderTrack />
						<ColorPicker.ChannelSliderThumb />
					</ColorPicker.ChannelSlider>
				</ColorPicker.Content>
			</ColorPicker.Positioner>
		</ColorPicker.Root>
	)
}
```

## Controlled State

```tsx
import { useState } from "react"

export default function ControlledExample() {
	const [color, setColor] = useState("#10b981")
	
	return (
		<div>
			<p>Selected color: {color}</p>
			<ColorPicker.Root value={color} onValueChange={setColor}>
				<ColorPicker.Control>
					<ColorPicker.ChannelInput channel="hex" />
					<ColorPicker.Trigger>
						<ColorPicker.Swatch />
					</ColorPicker.Trigger>
				</ColorPicker.Control>

	)
}
```

## Preset Swatches

```tsx
export default function SwatchesExample() {
	const presetColors = [
		"#ef4444", "#f97316", "#eab308", "#22c55e", 
		"#3b82f6", "#8b5cf6", "#ec4899", "#6b7280"
	]
	
	return (
		<ColorPicker.Root>
			<ColorPicker.Control>
				<ColorPicker.Trigger>
					<ColorPicker.Swatch />
				</ColorPicker.Trigger>
			</ColorPicker.Control>
			
			<ColorPicker.Positioner>
				<ColorPicker.Content>
					<ColorPicker.Area>
						<ColorPicker.AreaBackground />
						<ColorPicker.AreaThumb />
					</ColorPicker.Area>
					
					<div style={{ display: "grid", gridTemplateColumns: "repeat(4, 1fr)", gap: "0.5rem" }}>
						{presetColors.map(color => (
							<ColorPicker.SwatchTrigger key={color} value={color}>
								<ColorPicker.Swatch value={color} />
							</ColorPicker.SwatchTrigger>
						))}
					</div>
				</ColorPicker.Content>
			</ColorPicker.Positioner>
		</ColorPicker.Root>
	)
}
```

## API Reference

### Root Props
- `value`: `string` - Current color value
- `defaultValue`: `string` - Default color (uncontrolled)
- `onValueChange`: `(value: string) => void` - Color change handler
- `format`: `"hex" | "rgb" | "hsl"` - Output format

### Accessibility
- Full keyboard navigation
- Screen reader announcements for color values
- High contrast focus indicators
- ARIA labels for all interactive elements
